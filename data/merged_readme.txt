merged_training.txt

Format:
#image path #b1 b2 b3 b4 #x1 y2 … x5 y5 #gender #smile #wearing glasses #head pose

--image path: the image path
--b1 b2 b3 b4: the bounding box
--x1 y2 … x5 y5: the coordinates of the facial landmarks
--gender: 1 for male, 2 for female
--smile: 1 for smiling, 2 for not smiling
--glasses: 1 for wearing glasses, 2 for not wearing glasses.
--head pose: 1 for left profile, 2 for left, 3 for frontal, 4 for right, 5 for right profile
